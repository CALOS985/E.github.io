DEF变量设置

; player信息
[Info]
name = "Player" ;人物名字
displayname = "Player" ;人物显示的名字
versiondate = 09,09,2009 ;人物版本日期 (MM-DD-YYYY或X.XX)
mugenversion = 1.0 ;人物运行的MUGEN版本(X.XX)
author = "My Name" ;人物作者名
pal.defaults = 1,2,3,4 ;默认的色表优先顺序 ,从1开始编号
localcoord = 320,240 ;本地坐标空间宽和高

; player文件
[Files]
cmd = player.cmd ;设置Command文件
cns = player.cns ;设置Constants文件
st = player.cns ;设置States文件
stcommon = common1.cns ;公共状态 (从 data/ 或 motif)
sprite = player.sff ;图像资源素材
anim = player.air ;动画文件
sound = ;声音文件(没有就别设)
ai = kfm.ai ;AI文件

;将选择键与色表映射
[Palette Keymap]
x = 1 ;按X键选择色表1,等等
y = 2
z = 3
a = 4
b = 5
c = 6
;x2 = 7 ;按住start再按X键选择色表7,等等
;y2 = 8
;z2 = 9
;a2 = 10
;b2 = 11
;c2 = 12
